function fontChange(f) {
	if (typeof(Storage) !== "undefined") {
    var prev = localStorage.getItem("fontFamily");
	  localStorage.setItem("fontFamily", f);
	  document.body.style.fontFamily = localStorage.getItem("fontFamily");
    showFontChange(f, prev);
	}
}

function copyToClipboard(element) {
  var $temp = $("<input>");
  $("body").append($temp);
  $temp.val($(element).text()).select();
  document.execCommand("copy");
  $temp.remove();

  showCopied();
}

function showCopied() {
  var x = document.getElementById("snackbar1");
  x.className = "show";
  setTimeout(function(){ x.className = x.className.replace("show", ""); }, 3000);
}

function showFontChange(f, prev) {
  var x = document.getElementById("snackbar2");
  if (prev == f) {
    x.innerHTML = "site font is already ".concat(prev).concat("!");
  }
  else {
    x.innerHTML = "changed site font to: ".concat(f).concat("!");
  }
  x.className = "show";
  setTimeout(function(){ x.className = x.className.replace("show", ""); }, 3000);
}

function fontChange(f) {
	if (typeof(Storage) !== "undefined") {
    var prev = localStorage.getItem("fontFamily");
	  localStorage.setItem("fontFamily", f);
	  document.body.style.fontFamily = localStorage.getItem("fontFamily");
    showFontChange(f, prev);
	}
}

function copyToClipboard(element) {
  var $temp = $("<input>");
  $("body").append($temp);
  $temp.val($(element).text()).select();
  document.execCommand("copy");
  $temp.remove();

  showCopied();
}

function showCopied() {
  var x = document.getElementById("snackbar1");
  x.className = "show";
  setTimeout(function(){ x.className = x.className.replace("show", ""); }, 3000);
}

function showFontChange(f, prev) {
  var x = document.getElementById("snackbar2");
  if (prev == f) {
    x.innerHTML = "site font is already ".concat(prev).concat("!");
  }
  else {
    x.innerHTML = "changed site font to: ".concat(f).concat("!");
  }
  x.className = "show";
  setTimeout(function(){ x.className = x.className.replace("show", ""); }, 3000);
}
